Secure Password Manager

Project Overview

In this project, you'll build a secure password manager that stores passwords encrypted with a master password. This combines cryptography, secure data storage, and user interface design to create a practical security tool.

Why This Project Matters

Password reuse is a major security risk. According to security reports, 81% of data breaches involve weak or reused passwords. A secure password manager helps users maintain unique, strong passwords for each service. By building this tool, you'll learn:

  • Master password implementation and key derivation
  • Secure password database design
  • Password generation algorithms
  • User authentication and session management
  • Data encryption and secure storage practices

What You'll Build

A comprehensive password management system that:

  • Uses a master password to encrypt/decrypt the password database
  • Generates strong, unique passwords for different services
  • Stores passwords securely with AES-256 encryption
  • Provides a command-line interface for password operations
  • Includes password strength checking and security auditing
  • Supports secure backup and export functionality

Prerequisites

Before starting, make sure you have:

  • Python 3.8 or higher installed
  • Understanding of basic cryptography concepts
  • Familiarity with file operations and data structures
  • Knowledge of command-line interfaces

Build the Cryptographic Engine

Create the core encryption and decryption functionality.

from cryptography.hazmat.primitives import hashes
from cryptography.hazmat.primitives.kdf.pbkdf2 import PBKDF2HMAC
from cryptography.hazmat.primitives.ciphers import Cipher, algorithms, modes
from cryptography.hazmat.backends import default_backend
from cryptography.exceptions import InvalidKey
import secrets
import base64
import os

class CryptoEngine:
    """Handles encryption and decryption operations"""

    def __init__(self):
        self.backend = default_backend()

    def derive_key(self, password, salt=None, iterations=100000):
        """
        Derive encryption key from password using PBKDF2

        Args:
            password (str): Master password
            salt (bytes): Salt for key derivation
            iterations (int): Number of PBKDF2 iterations

        Returns:
            tuple: (derived_key, salt_used)
        """
        if salt is None:
            salt = secrets.token_bytes(32)

        kdf = PBKDF2HMAC(
            algorithm=hashes.SHA256(),
            length=32,
            salt=salt,
            iterations=iterations,
            backend=self.backend
        )

        key = kdf.derive(password.encode('utf-8'))
        return key, salt

    def encrypt_data(self, data, password):
        """
        Encrypt data using AES-256-GCM

        Args:
            data (str): Data to encrypt
            password (str): Encryption password

        Returns:
            dict: Encrypted data with metadata
        """
        key, salt = self.derive_key(password)
        nonce = secrets.token_bytes(12)  # 96-bit nonce for GCM

        cipher = Cipher(algorithms.AES(key), modes.GCM(nonce), backend=self.backend)
        encryptor = cipher.encryptor()

        ciphertext = encryptor.update(data.encode('utf-8')) + encryptor.finalize()
        tag = encryptor.tag

        return {
            'ciphertext': base64.b64encode(ciphertext).decode('utf-8'),
            'salt': base64.b64encode(salt).decode('utf-8'),
            'nonce': base64.b64encode(nonce).decode('utf-8'),
            'tag': base64.b64encode(tag).decode('utf-8'),
            'iterations': 100000
        }

    def decrypt_data(self, encrypted_data, password):
        """
        Decrypt data using AES-256-GCM

        Args:
            encrypted_data (dict): Encrypted data with metadata
            password (str): Decryption password

        Returns:
            str: Decrypted data

        Raises:
            ValueError: If decryption fails
        """
        try:
            # Decode base64 encoded components
            ciphertext = base64.b64decode(encrypted_data['ciphertext'])
            salt = base64.b64decode(encrypted_data['salt'])
            nonce = base64.b64decode(encrypted_data['nonce'])
            tag = base64.b64decode(encrypted_data['tag'])
            iterations = encrypted_data.get('iterations', 100000)

            # Derive key
            key, _ = self.derive_key(password, salt, iterations)

            # Create cipher for decryption
            cipher = Cipher(algorithms.AES(key), modes.GCM(nonce, tag), backend=self.backend)
            decryptor = cipher.decryptor()

            # Decrypt data
            plaintext = decryptor.update(ciphertext) + decryptor.finalize()

            return plaintext.decode('utf-8')

        except (InvalidKey, ValueError, KeyError) as e:
            raise ValueError(f"Decryption failed: {e}")

    def verify_password(self, stored_hash, password):
        """
        Verify password against stored hash

        Args:
            stored_hash (str): Stored password hash
            password (str): Password to verify

        Returns:
            bool: True if password is correct
        """
        import bcrypt
        return bcrypt.checkpw(password.encode('utf-8'), stored_hash.encode('utf-8'))

    def hash_password(self, password):
        """
        Hash password for storage

        Args:
            password (str): Password to hash

        Returns:
            str: Hashed password
        """
        import bcrypt
        return bcrypt.hashpw(password.encode('utf-8'), bcrypt.gensalt()).decode('utf-8')

Cryptographic choices: AES-256-GCM provides both confidentiality and authenticity, while PBKDF2 protects against brute-force attacks.

Create the Password Generator

Build a secure password generation system with customizable options.

import secrets
import string

class PasswordGenerator:
    """Generates secure passwords with various options"""

    def __init__(self):
        self.lowercase = string.ascii_lowercase
        self.uppercase = string.ascii_uppercase
        self.digits = string.digits
        self.symbols = "!@#$%^&*()_+-=[]{}|;:,.<>?"

    def generate_password(self, length=16, use_uppercase=True, use_digits=True,
                         use_symbols=True, exclude_similar=False):
        """
        Generate a secure password

        Args:
            length (int): Password length
            use_uppercase (bool): Include uppercase letters
            use_digits (bool): Include digits
            use_symbols (bool): Include symbols
            exclude_similar (bool): Exclude similar characters (1,l,I,0,O)

        Returns:
            str: Generated password
        """
        # Build character set
        chars = self.lowercase

        if use_uppercase:
            chars += self.uppercase
        if use_digits:
            chars += self.digits
        if use_symbols:
            chars += self.symbols

        if exclude_similar:
            # Remove similar characters
            similar = '1lI0O'
            chars = ''.join(c for c in chars if c not in similar)

        if not chars:
            raise ValueError("No character sets selected")

        # Ensure at least one character from each selected set
        password = []

        if use_uppercase and self.uppercase:
            password.append(secrets.choice(self.uppercase))
        if use_digits and self.digits:
            password.append(secrets.choice(self.digits))
        if use_symbols and self.symbols:
            password.append(secrets.choice(self.symbols))

        # Fill remaining length
        remaining_length = length - len(password)
        if remaining_length > 0:
            password.extend(secrets.choice(chars) for _ in range(remaining_length))

        # Shuffle the password
        password_list = list(password)
        secrets.SystemRandom().shuffle(password_list)

        return ''.join(password_list)

    def calculate_entropy(self, password):
        """
        Calculate password entropy (bits)

        Args:
            password (str): Password to analyze

        Returns:
            float: Password entropy in bits
        """
        char_sets = 0
        length = len(password)

        if any(c in self.lowercase for c in password):
            char_sets += 26
        if any(c in self.uppercase for c in password):
            char_sets += 26
        if any(c in self.digits for c in password):
            char_sets += 10
        if any(c in self.symbols for c in password):
            char_sets += len(self.symbols)

        if char_sets == 0:
            return 0

        # Entropy = log2(char_sets^length)
        import math
        entropy = length * math.log2(char_sets)

        return round(entropy, 2)

    def assess_strength(self, password):
        """
        Assess password strength

        Args:
            password (str): Password to assess

        Returns:
            dict: Strength assessment
        """
        entropy = self.calculate_entropy(password)
        length = len(password)

        if entropy >= 100 and length >= 16:
            strength = "Very Strong"
            score = 5
        elif entropy >= 80 and length >= 12:
            strength = "Strong"
            score = 4
        elif entropy >= 60 and length >= 10:
            strength = "Good"
            score = 3
        elif entropy >= 40 and length >= 8:
            strength = "Fair"
            score = 2
        else:
            strength = "Weak"
            score = 1

        return {
            'strength': strength,
            'score': score,
            'entropy': entropy,
            'length': length
        }

Password generation: Using cryptographically secure random number generation ensures truly random passwords.

Implement the Password Database

Create a secure system for storing and managing encrypted password entries.

import json
import os
from datetime import datetime
from crypto_engine import CryptoEngine

class PasswordDatabase:
    """Manages encrypted password storage"""

    def __init__(self, db_file='passwords.db'):
        self.db_file = db_file
        self.crypto = CryptoEngine()
        self.master_hash = None
        self.is_unlocked = False

    def initialize_database(self, master_password):
        """
        Initialize a new password database

        Args:
            master_password (str): Master password for the database
        """
        self.master_hash = self.crypto.hash_password(master_password)

        # Create empty database structure
        empty_db = {
            'metadata': {
                'created': datetime.now().isoformat(),
                'version': '1.0',
                'master_hash': self.master_hash
            },
            'passwords': {}
        }

        # Save unencrypted metadata (master hash is already hashed)
        with open(self.db_file, 'w') as f:
            json.dump(empty_db, f, indent=2)

        self.is_unlocked = True
        print("Password database initialized successfully")

    def unlock_database(self, master_password):
        """
        Unlock the password database

        Args:
            master_password (str): Master password

        Returns:
            bool: True if unlock successful
        """
        if not os.path.exists(self.db_file):
            print("Database file not found. Initialize first.")
            return False

        try:
            with open(self.db_file, 'r') as f:
                db_data = json.load(f)

            stored_hash = db_data['metadata']['master_hash']

            if self.crypto.verify_password(stored_hash, master_password):
                self.master_hash = stored_hash
                self.is_unlocked = True
                return True
            else:
                print("Incorrect master password")
                return False

        except (json.JSONDecodeError, KeyError) as e:
            print(f"Database corruption or invalid format: {e}")
            return False

    def lock_database(self):
        """Lock the database"""
        self.is_unlocked = False
        self.master_hash = None

    def add_password(self, service, username, password, master_password, notes=""):
        """
        Add a password entry to the database

        Args:
            service (str): Service name (e.g., 'gmail.com')
            username (str): Username or email
            password (str): Password to store
            master_password (str): Master password for encryption
            notes (str): Optional notes
        """
        if not self.is_unlocked:
            raise ValueError("Database is locked")

        # Load current database
        with open(self.db_file, 'r') as f:
            db_data = json.load(f)

        # Create password entry
        entry = {
            'username': username,
            'password': password,
            'notes': notes,
            'created': datetime.now().isoformat(),
            'modified': datetime.now().isoformat()
        }

        # Encrypt the sensitive data
        sensitive_data = {
            'password': password,
            'notes': notes
        }

        encrypted_data = self.crypto.encrypt_data(json.dumps(sensitive_data), master_password)

        # Store encrypted entry
        db_data['passwords'][service] = {
            'username': username,
            'encrypted_data': encrypted_data,
            'created': entry['created'],
            'modified': entry['modified']
        }

        # Save updated database
        with open(self.db_file, 'w') as f:
            json.dump(db_data, f, indent=2)

    def get_password(self, service, master_password):
        """
        Retrieve a password from the database

        Args:
            service (str): Service name
            master_password (str): Master password for decryption

        Returns:
            dict: Password entry or None if not found
        """
        if not self.is_unlocked:
            raise ValueError("Database is locked")

        with open(self.db_file, 'r') as f:
            db_data = json.load(f)

        if service not in db_data['passwords']:
            return None

        entry = db_data['passwords'][service]

        # Decrypt sensitive data
        try:
            decrypted_data = self.crypto.decrypt_data(entry['encrypted_data'], master_password)
            sensitive_data = json.loads(decrypted_data)

            return {
                'service': service,
                'username': entry['username'],
                'password': sensitive_data['password'],
                'notes': sensitive_data['notes'],
                'created': entry['created'],
                'modified': entry['modified']
            }
        except ValueError:
            raise ValueError("Failed to decrypt password - incorrect master password")

    def list_services(self):
        """
        List all stored services

        Returns:
            list: List of service names
        """
        if not self.is_unlocked:
            raise ValueError("Database is locked")

        with open(self.db_file, 'r') as f:
            db_data = json.load(f)

        return list(db_data['passwords'].keys())

    def delete_password(self, service):
        """
        Delete a password entry

        Args:
            service (str): Service name to delete
        """
        if not self.is_unlocked:
            raise ValueError("Database is locked")

        with open(self.db_file, 'r') as f:
            db_data = json.load(f)

        if service in db_data['passwords']:
            del db_data['passwords'][service]

            with open(self.db_file, 'w') as f:
                json.dump(db_data, f, indent=2)
        else:
            raise ValueError(f"Service '{service}' not found")

Database security: Only metadata is stored unencrypted; all sensitive password data is encrypted with the master password.

Build the Command-Line Interface

Create a user-friendly command-line interface for password management operations.

#!/usr/bin/env python3
"""
Secure Password Manager
A command-line password management system with encryption
"""

import argparse
import getpass
import sys
from password_generator import PasswordGenerator
from database_manager import PasswordDatabase

def get_master_password():
    """Securely get master password from user"""
    return getpass.getpass("Enter master password: ")

def confirm_master_password():
    """Get and confirm master password"""
    while True:
        password = getpass.getpass("Enter master password: ")
        confirm = getpass.getpass("Confirm master password: ")

        if password == confirm:
            return password
        else:
            print("Passwords do not match. Try again.")

def init_command(args):
    """Initialize a new password database"""
    if args.database.exists():
        if not args.force:
            response = input(f"Database {args.database} already exists. Overwrite? (y/N): ")
            if response.lower() != 'y':
                return

    master_password = confirm_master_password()
    db = PasswordDatabase(str(args.database))
    db.initialize_database(master_password)
    print(f"Password database initialized: {args.database}")

def add_command(args):
    """Add a password entry"""
    master_password = get_master_password()
    db = PasswordDatabase(str(args.database))

    if not db.unlock_database(master_password):
        sys.exit(1)

    # Get password (generate or manual)
    if args.generate:
        generator = PasswordGenerator()
        password = generator.generate_password(
            length=args.length,
            use_uppercase=not args.no_uppercase,
            use_digits=not args.no_digits,
            use_symbols=not args.no_symbols
        )
        print(f"Generated password: {password}")

        # Assess strength
        assessment = generator.assess_strength(password)
        print(f"Strength: {assessment['strength']} (entropy: {assessment['entropy']} bits)")
    else:
        password = getpass.getpass("Enter password: ")
        confirm = getpass.getpass("Confirm password: ")
        if password != confirm:
            print("Passwords do not match")
            sys.exit(1)

    # Add to database
    notes = args.notes if args.notes else ""
    db.add_password(args.service, args.username, password, master_password, notes)
    print(f"Password added for service: {args.service}")

def get_command(args):
    """Retrieve a password"""
    master_password = get_master_password()
    db = PasswordDatabase(str(args.database))

    if not db.unlock_database(master_password):
        sys.exit(1)

    entry = db.get_password(args.service, master_password)
    if entry:
        print(f"Service: {entry['service']}")
        print(f"Username: {entry['username']}")
        print(f"Password: {entry['password']}")
        if entry['notes']:
            print(f"Notes: {entry['notes']}")
        print(f"Created: {entry['created']}")
    else:
        print(f"No password found for service: {args.service}")

def list_command(args):
    """List all services"""
    master_password = get_master_password()
    db = PasswordDatabase(str(args.database))

    if not db.unlock_database(master_password):
        sys.exit(1)

    services = db.list_services()
    if services:
        print("Stored services:")
        for service in sorted(services):
            print(f"  ‚Ä¢ {service}")
    else:
        print("No passwords stored")

def generate_command(args):
    """Generate a password"""
    generator = PasswordGenerator()
    password = generator.generate_password(
        length=args.length,
        use_uppercase=not args.no_uppercase,
        use_digits=not args.no_digits,
        use_symbols=not args.no_symbols
    )

    assessment = generator.assess_strength(password)

    print(f"Generated Password: {password}")
    print(f"Length: {assessment['length']}")
    print(f"Entropy: {assessment['entropy']} bits")
    print(f"Strength: {assessment['strength']}")

def main():
    parser = argparse.ArgumentParser(
        description="Secure Password Manager",
        formatter_class=argparse.RawDescriptionHelpFormatter,
        epilog="""
Examples:
  # Initialize database
  python password_manager.py init

  # Add password
  python password_manager.py add gmail user@gmail.com --generate

  # Get password
  python password_manager.py get gmail

  # Generate password
  python password_manager.py generate --length 20

  # List services
  python password_manager.py list
        """
    )

    parser.add_argument('--database', type=Path, default='passwords.db',
                       help='Database file path (default: passwords.db)')

    subparsers = parser.add_subparsers(dest='command', help='Available commands')

    # Init command
    init_parser = subparsers.add_parser('init', help='Initialize password database')
    init_parser.add_argument('--force', action='store_true',
                            help='Overwrite existing database')

    # Add command
    add_parser = subparsers.add_parser('add', help='Add password entry')
    add_parser.add_argument('service', help='Service name')
    add_parser.add_argument('username', help='Username or email')
    add_parser.add_argument('--generate', action='store_true',
                           help='Generate a secure password')
    add_parser.add_argument('--length', type=int, default=16,
                           help='Generated password length')
    add_parser.add_argument('--no-uppercase', action='store_true',
                           help='Exclude uppercase letters')
    add_parser.add_argument('--no-digits', action='store_true',
                           help='Exclude digits')
    add_parser.add_argument('--no-symbols', action='store_true',
                           help='Exclude symbols')
    add_parser.add_argument('--notes', help='Optional notes')

    # Get command
    get_parser = subparsers.add_parser('get', help='Retrieve password')
    get_parser.add_argument('service', help='Service name')

    # List command
    list_parser = subparsers.add_parser('list', help='List stored services')

    # Generate command
    gen_parser = subparsers.add_parser('generate', help='Generate secure password')
    gen_parser.add_argument('--length', type=int, default=16,
                           help='Password length (default: 16)')
    gen_parser.add_argument('--no-uppercase', action='store_true',
                           help='Exclude uppercase letters')
    gen_parser.add_argument('--no-digits', action='store_true',
                           help='Exclude digits')
    gen_parser.add_argument('--no-symbols', action='store_true',
                           help='Exclude symbols')

    args = parser.parse_args()

    if not args.command:
        parser.print_help()
        return

    # Execute command
    if args.command == 'init':
        init_command(args)
    elif args.command == 'add':
        add_command(args)
    elif args.command == 'get':
        get_command(args)
    elif args.command == 'list':
        list_command(args)
    elif args.command == 'generate':
        generate_command(args)

if __name__ == "__main__":
    main()
